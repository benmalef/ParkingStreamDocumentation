import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';

@Component({
  selector: 'app-log-out',
  templateUrl: './log-out.component.html',
  styleUrls: ['./log-out.component.css']
})
export class LogOutComponent implements OnInit {

  constructor(private router: Router) { }

  ngOnInit(): void {
  }

  onClickAccept() {
    this.router.navigate(["/login"]);
  }
  onClickReject() {
    this.router.navigate(["/main/welcomePage"]);
  }
}

<div class="text-center">
    <div class="bg-secondary p-5">

        <button (click)="onClickAccept()" type="button" class="btn  btn-lg m-2 btn-light rounded">Yes</button>
        <button (click)="onClickReject()" type="button" class="btn btn-light btn-lg m-2 rounded"> No</button>

</div>
